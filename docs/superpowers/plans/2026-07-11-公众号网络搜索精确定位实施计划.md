# 公众号网络搜索精确定位 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

**Goal:** 让公众号 RPA 在本地会话定位失败后，通过微信顶部“搜索网络结果”定位名称完全等于配置值的公众号，并保持 No-Go 门禁直到只读 probe 通过。

**Architecture:** 保留 ChatWith、本地搜索和窗口就绪等待作为第一层；新增网络搜索作为最后的定位分支。网络分支只点击标准化后完全匹配的结果，点击后仍必须通过既有目标窗口校验，才能进入文章历史取链。

**Tech Stack:** Python 3.12、wxauto4、pywinauto UIA、pytest、现有 WxautoArticleRpaClient。

## Global Constraints

- 只接受名称完全等于 account_name 的网络搜索结果；禁止模糊匹配、首项回车和 force=True。
- 网络搜索失败时不得写入采集数据、不得进入历史文章点击、不得输出搜索结果名称或文章 URL。
- 不新增 RSS、数据库表、前端页面或调度频率变更。
- 真实验证仅执行只读 article-rpa-probe；重跑单目标 POC 需要管理员在 probe 通过后再次确认。
- 所有中文文档使用 UTF-8 无 BOM，变更后执行编码校验。

~~~python
_NETWORK_SEARCH_ENTRY_TEXT = "搜索网络结果"
_NETWORK_RESULT_CLASS_NAMES = {"mmui::SearchContentCellView", "mmui::XTableCell"}


def _find_network_search_entry(main_window: Any) -> Any | None:
    for control in _safe_descendants(main_window, limit=1200):
        if _normalize_match_text(_safe_text(control)) != _NETWORK_SEARCH_ENTRY_TEXT:
            continue
        if _safe_control_type(control) in {"ListItem", "Button"}:
            return control
    return None


def _find_exact_network_search_result(container: Any, account_name: str) -> Any | None:
    expected = _normalize_match_text(account_name)
    for control in _safe_descendants(container, limit=3000):
        if _safe_class_name(control) in _NETWORK_RESULT_CLASS_NAMES and _normalize_match_text(_safe_text(control)) == expected:
            return control
    return None
~~~

~~~python
def open_public_account(self, account_name: str) -> None:
    _chat_with_retry(self.wx, account_name)
    if self.open_account_search_fallback_enabled and not self._wait_for_public_account_window(account_name):
        if self._open_public_account_from_main_search(account_name) and self._wait_for_public_account_window(account_name):
            self.current_account_name = account_name
            return
        if not self._open_public_account_from_network_search(account_name):
            raise RuntimeError("public account window not found")
        if not self._wait_for_public_account_window(account_name):
            raise RuntimeError("public account window not found")
    self.current_account_name = account_name
~~~

~~~python
def _wait_for_network_search_entry(main_window: Any, timeout_seconds: float = 4.0) -> Any | None:
    deadline = time.monotonic() + timeout_seconds
    while time.monotonic() < deadline:
        entry = _find_network_search_entry(main_window)
        if entry is not None:
            return entry
        time.sleep(0.2)
    return None


def _wait_for_exact_network_search_result(
    container: Any,
    account_name: str,
    timeout_seconds: float = 4.0,
) -> Any | None:
    deadline = time.monotonic() + timeout_seconds
    while time.monotonic() < deadline:
        result = _find_exact_network_search_result(container, account_name)
        if result is not None:
            return result
        time.sleep(0.2)
    return None


def _open_public_account_from_network_search(self, account_name: str) -> bool:
    pyperclip = importlib.import_module("pyperclip")
    pywinauto = importlib.import_module("pywinauto")
    keyboard = importlib.import_module("pywinauto.keyboard")
    main_window = _find_wechat_main_window(pywinauto.Desktop(backend="uia"))
    search_edit = None if main_window is None else _find_main_search_edit(main_window)
    if search_edit is None:
        return False
    previous_clipboard = pyperclip.paste()
    try:
        _focus_window(main_window)
        _activate_control(search_edit, prefer_physical=True)
        pyperclip.copy(account_name)
        keyboard.send_keys("^a{BACKSPACE}^v")
        entry = _wait_for_network_search_entry(main_window)
        if entry is None:
            return False
        _activate_control(entry, prefer_physical=True)
        result = _wait_for_exact_network_search_result(main_window, account_name)
        if result is None:
            return False
        _activate_control(result, prefer_physical=True)
        return True
    finally:
        pyperclip.copy(previous_clipboard)
~~~

The method must not press Enter to choose the first result or log result text.
